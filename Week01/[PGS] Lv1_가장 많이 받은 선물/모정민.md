import java.io.*;
import java.util.*;

class Solution {
    
    public int solution(String[] friends, String[] gifts) {
        
        // ë³€ìˆ˜ ì´ˆê¸°í™”
        int n = friends.length;
        HashMap<String, Integer> map = new HashMap<>();
        for(int i = 0; i < n; i++){
            map.put(friends[i], i);
        }
        
        // ëˆ„ê°€ -> ëˆ„êµ¬ì—ê²Œ ì„ ë¬¼ì„ ì¤¬ëŠ”ì§€ ê¸°ë¡
        int[][] list = new int[n][n];
        
        // ì„ ë¬¼ ì§€ìˆ˜
        int[] present = new int[n];
        
        // ë‹¤ìŒ ë‹¬ ì„ ë¬¼ì„ ëª‡ë²ˆ ë°›ì•„ì•¼ í•˜ëŠ”ì§€ ê¸°ë¡
        int[] count = new int[n];
        
        int max = Integer.MIN_VALUE;
        
    
        for(String g : gifts){
            
            String[] split = g.split(" ");
            
            int from = map.get(split[0]);
            int to = map.get(split[1]);
            
            // ì„ ë¬¼ì„ ì¤€ ì‚¬ëŒì„ ì¹´ìš´íŒ…
            list[from][to]++;
            
            // ì„ ë¬¼ ì§€ìˆ˜ ê³„ì‚°
            present[from]++;
            present[to]--;
        }
        
        for(int i = 0; i < n; i++){
            for(int j = i+1; j < n; j++){
                
                // ì„ ë¬¼ì„ ë” ë§ì´ ì¤€ ì‚¬ëŒì´ ì„ ë¬¼ì„ ë°›ì•„ì•¼ í•¨
                if(list[i][j] > list[j][i]){
                    count[i]++;
                }
                else if(list[i][j] < list[j][i]){
                    count[j]++;
                }
                
                // ë‘˜ì´ ì£¼ê³  ë°›ì€ íšŸìˆ˜ê°€ ê°™ë‹¤ë©´
                else{
                    // ì„ ë¬¼ ì§€ìˆ˜ ë¹„êµ
                    if(present[i] > present[j]){
                        count[i]++;
                    }else if(present[i] < present[j]){
                        count[j]++;
                    } 
                }
            }
        }
        
        for(int c : count){
            max = Math.max(max, c);
        }
        
        
        return max;
    }

}
