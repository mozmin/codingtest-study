import java.util.*;
class Solution {
    public int solution(int n, int[] stations, int w) {
        int count = 0;
        int w_size = w * 2 + 1;
        int[] table = new int[n+1];
        for(int i = 0 ; i < stations.length; i++){
            int ind = stations[i];
            for(int j = ind - w; j <= ind + w; j++){
                if(j < 0 || j > n) continue;
                table[j] = 1;
            }
        }     
        
        int nn = 0;
        for(int i = 1 ; i < n+1; i++){
            if(table[i] == 1){
                if(nn == 0) continue;
                int num = nn / w_size;
                if(nn % w_size != 0) num += 1;
                count += num;
                nn = 0;
            }
            else if(table[i] == 0){
                nn++;
            }
        }
        if(nn != 0){
            int num = nn / w_size;
            if(nn % w_size != 0) num += 1;
            count += num;
        }
        

        return count;
    }
}
