import java.util.*;

class Solution {
    public String solution(String[] survey, int[] choices) {
        String answer = "";
        StringBuilder sb = new StringBuilder();
        //ì„±ê²© ìœ í˜• ì§€í‘œ - R T C F J M A N
        int[] score = new int[8];
        char[] chars = {'R', 'T', 'C', 'F', 'J', 'M', 'A', 'N'};
        
        for(int i = 0; i < survey.length; i++){
            
            int choice = choices[i];
            if(choice == 4) continue;
            
            String sur = survey[i];
            Character s1 = sur.charAt(0);
            Character s2 = sur.charAt(1);
            
            int s1_num = -1; 
            int s2_num = -1;
            for(int j = 0; j < 8; j++){
                if(s1 == chars[j]){
                    s1_num = j;
                    break;
                }
            }
            if(s1_num % 2 == 0) s2_num = s1_num + 1;
            else s2_num = s1_num - 1;
            
            if (choice < 4){
                choice = 4 - choice;
                score[s1_num] += choice;
                
            }
            else{
                choice -= 4;
                score[s2_num] += choice;
            }       
        }

        for(int i = 0 ; i < 4; i++){
            if(score[i*2] >= score[i*2+1]){
                sb.append(chars[i*2]);
            }
            else{
                sb.append(chars[i*2+1]);
            }
        }
        
        answer = sb.toString();        
        return answer;
    }
}
