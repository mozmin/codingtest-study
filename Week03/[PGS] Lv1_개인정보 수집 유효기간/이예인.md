import java.util.*;

class Solution {
    public int[] solution(String today, String[] terms, String[] privacies) {
        
        int today_year = Integer.parseInt(today.substring(0,4));
        int today_month = Integer.parseInt(today.substring(5,7));
        int today_day = Integer.parseInt(today.substring(8));
        
        int count = 0;
        int[] ans = new int[privacies.length];

        for(int i = 0; i < privacies.length; i++){
            String privacy = privacies[i];
            int privacy_year = Integer.parseInt(privacy.substring(0,4));
            int privacy_month = Integer.parseInt(privacy.substring(5,7));
            int privacy_day = Integer.parseInt(privacy.substring(8,10));
            char privacy_char = privacy.charAt(11);
            
            int term_month = -1;
            
            for(int j = 0; j < terms.length; j++){
                String term = terms[j];
                if(term.charAt(0) == privacy_char){
                    term_month = Integer.parseInt(term.substring(2));
                    break;
                }
            }
            
            privacy_month += term_month;
            if(privacy_month > 12){
                privacy_year += privacy_month / 12;
                privacy_month = privacy_month % 12;
            }
            
            privacy_day -= 1;
            if(privacy_day == 0){
                privacy_month -= 1;
                privacy_day = 28;
            }
            
            if(privacy_month < 1){
                privacy_year -= 1;
                privacy_month = 12;
            }
            
            if(privacy_year < today_year) {
                count++;
                ans[i] = 1;
            }
            else if(privacy_year == today_year){
                if(privacy_month < today_month){
                    count++;
                    ans[i] = 1;
                }
                else if(privacy_month == today_month && privacy_day < today_day){
                    count++;
                    ans[i] = 1;
                }
            } 
            
        }

        int[] answer = new int[count];
        int k = 0;
        for(int i = 0 ; i < ans.length; i++){
            if(ans[i] == 1){
                answer[k] = i+1;
                k++;
            }
        }
                                                                             
        return answer;
    }
}
