import java.util.*;

class Solution {

    // 최대 감염 수 저장
    int answer = 0;

    // 최대 행동 횟수
    int maxK;

    public int solution(int n, int infection, int[][] edges, int k) {

        maxK = k;

        // 그래프 생성
        List<int[]>[] graph = new ArrayList[n + 1];

        for (int i = 1; i <= n; i++) {
            graph[i] = new ArrayList<>();
        }

        // 그래프 연결
        for (int[] edge : edges) {

            int from = edge[0];
            int to = edge[1];
            int type = edge[2];

            graph[from].add(new int[]{to, type});
            graph[to].add(new int[]{from, type});
        }

        // 감염 여부 배열
        boolean[] infected = new boolean[n + 1];

        // 시작 감염 노드
        infected[infection] = true;

        // DFS 시작
        dfs(graph, infected, 0);

        return answer;
    }

    // 행동 순서를 전부 탐색하는 DFS
    public void dfs(List<int[]>[] graph,
                    boolean[] infected,
                    int depth) {

        // 현재 감염 수 계산
        answer = Math.max(answer, countInfected(infected));

        // 행동 횟수 다 사용하면 종료
        if (depth == maxK) {
            return;
        }

        // A, B, C 타입 전부 시도
        for (int type = 1; type <= 3; type++) {

            // 현재 감염 상태 복사
            boolean[] nextInfected = infected.clone();

            // 해당 타입 파이프 열어서 감염 퍼뜨리기
            spread(graph, nextInfected, type);

            // 다음 행동으로 진행
            dfs(graph, nextInfected, depth + 1);
        }
    }

    // 특정 타입 파이프를 열었을 때 감염 확산
    public void spread(List<int[]>[] graph,
                       boolean[] infected,
                       int openType) {

        Queue<Integer> queue = new ArrayDeque<>();

        // 현재 감염된 노드들을 큐에 삽입
        for (int i = 1; i < infected.length; i++) {
            if (infected[i]) {
                queue.offer(i);
            }
        }

        // BFS 시작
        while (!queue.isEmpty()) {

            int now = queue.poll();

            for (int[] next : graph[now]) {

                int nextNode = next[0];
                int pipeType = next[1];

                // 같은 타입 파이프이고 아직 감염 안 됐으면
                if (pipeType == openType && !infected[nextNode]) {

                    infected[nextNode] = true;

                    queue.offer(nextNode);
                }
            }
        }
    }

    // 감염된 개수 세기
    public int countInfected(boolean[] infected) {

        int count = 0;

        for (int i = 1; i < infected.length; i++) {
            if (infected[i]) {
                count++;
            }
        }

        return count;
    }
}
